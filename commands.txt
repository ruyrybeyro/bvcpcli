# Example: provision a Windows 11 VM named "w1"
# Lines starting with # are comments and are ignored by bvcpcli

# Create the VM using the w1 template
vm create w1 w1

# CPU and architecture
vm set w1 arch amd
vm set w1 sys   win
vm set w1 cpu.socket 1
vm set w1 cpu.core   4

# Memory
vm set w1 memory 16G

# VNC display settings
vm set w1 vnc.display 1024_768
vm set w1 vnc.wait    false

# Use local time (recommended for Windows guests)
vm set w1 utc_time    false

# Attach Windows 11 installation ISO (unattended install with VirtIO drivers baked in)
vm cdrom attach w1 Win11_25H2_English_x64_virtio.iso

# Create and attach a 100G NVMe disk on the zstor pool
vm disk create w1 /zstor w1disk 100G
vm disk attach w1 w1_w1disk.img w1_w1disk auto nvme

# Add a VirtIO NIC on the private network
vm nic add_virtio w1 privnet

# Boot the VM
vm start w1
